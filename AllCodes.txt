        {
            clsTask SelectedTask = GetSelectedTask();
            if (SelectedTask == null) return;
            string timeTemplate = "00:00:00";

            if (SelectedTask != null)
            {
                TimeSpan t = TimeSpan.FromSeconds(SelectedTask.TotalTime);
                timeTemplate = string.Format("{0:D2}:{1:D2}:{2:D2}", (int)t.TotalHours, t.Minutes, t.Seconds);
            }
            txtDetials.AppendText(Environment.NewLine + "Task Name : " + SelectedTask.Name);
            txtDetials.AppendText(Environment.NewLine + "Task Description : " + SelectedTask.Description);
            txtDetials.AppendText(Environment.NewLine + "Task Time : " + timeTemplate);
        }
        void RefreshProjectDetils()
        {
            clsProject SelectedProject = GetSelectedProject();

            txtDetials.Text = "Project Name : " + SelectedProject.Name;
            txtDetials.AppendText(Environment.NewLine + "Project Description : " + SelectedProject.Description);
            txtDetials.AppendText(Environment.NewLine + "Project Expected Time : " + SelectedProject.ExpectedTime + " H ");

        }
        void UpdateRemainingTime(clsProject SelectedProject)
        {

            int totalTasksSeconds = 0;

            foreach (clsTask task in SelectedProject.TasksList)
            {
                totalTasksSeconds += task.TotalTime;
            }

            int remainingSeconds = ProjectManager.GetProjectRemainingTime(SelectedProject);
            TimeSpan t = TimeSpan.FromSeconds(Math.Max(remainingSeconds, 0));

            if (remainingSeconds > 0)
                txtTotalProjectTime.Text = $"{(int)t.TotalHours}h {t.Minutes}m remaining";
            else
                txtTotalProjectTime.Text = "Overdue!";
            btnStart.Enabled = remainingSeconds > 0;

            txtTotalProjectTime.ForeColor = remainingSeconds >= 0 ? Color.Black : Color.Red;

            if (remainingSeconds <= 0)
                StartNotifications();
        }
        void RefreshDetials()
        {
            if (cbProjects.SelectedIndex == -1) return;

            clsProject SelectedProject = GetSelectedProject();
            if (SelectedProject != null)
            {
                RefreshProjectDetils();
                if (SelectedProject.TasksList != null && SelectedProject.TasksList.Count > 0)
                    RefreshTaskDetials();

                UpdateRemainingTime(SelectedProject);
            }

        }

        private void btnStope_Click(object sender, EventArgs e)

                return;

            EndTime = DateTime.Now;

Project Data :
using System;
using System.Collections.Generic;
using System.IO;

namespace Personal_Project_Time_Tracker.DAL
{
    public static class ProjectData
    {
        private const string FilePath = "Projects.txt";

        public static void SaveDataToFile(List<clsProject> projects)
        {
            using (StreamWriter writer = new StreamWriter(FilePath, false))
            {
                foreach (var project in projects)
                {
                    writer.WriteLine($"P#:#{project.ID}#:#{project.Name}#:#{project.Description}#:#{project.ExpectedTime}");

                    foreach (var task in project.TasksList)
                    {
                        writer.WriteLine($"T#:#{task.Name}#:#{task.Description}#:#{task.TotalTime}");
                    }
                }
            }
        }
        public static int GetNextProjectID(List<clsProject> ProjectsList)
        {
            int maxID = 0;

            foreach (clsProject project in ProjectsList)
            {
                if (project.ID > maxID)
                {
                    maxID = project.ID;
                }
            }

            return maxID + 1;
        }
        public static List<clsProject> LoadProjects()
        {
            List<clsProject> projects = new List<clsProject>();

            if (!File.Exists(FilePath))
                return projects;

            clsProject currentProject = null;

            foreach (var line in File.ReadAllLines(FilePath))
            {
                if (string.IsNullOrWhiteSpace(line)) continue;

                var parts = line.Split(new[] { "#:#" }, StringSplitOptions.None);

                if (parts[0] == "P")
                {
                    currentProject = new clsProject(int.Parse(parts[1]), parts[2], parts[3], int.Parse(parts[4]));

                    projects.Add(currentProject);
                }
                else if (parts[0] == "T" && currentProject != null)
                {
                    currentProject.TasksList.Add(new clsTask(parts[1], parts[2], int.Parse(parts[3])));
                }
            }

            return projects;
        }
    }
}

using System;
using System.Collections.Generic;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Personal_Project_Time_Tracker
{
    public static class ProjectManager
    { // THIS FILE IS BLL (BUSINESS LOGIC LAYER )
        // في هاد الملف لح يكون كل لوجيك كل حساب 
        /*
            1-2_ حساب الوقت المتبقي للمشورع و التحقق من اذا خلص وقتو
            3-4_ اضافة مشروع و مهمه للمشروع 
            5_ اضافة الوقت للمهمة
        */
        public static int GetProjectRemainingTime(clsProject project)
        {
            int totalTaskTime = 0;
            foreach (var t in project.TasksList)
                totalTaskTime += t.TotalTime;

            return project.ExpectedTime - totalTaskTime;
        }

        public static bool IsProjectOverdue(clsProject project)
        {
            return GetProjectRemainingTime(project) <= 0;
        }

        public static void AddTask(clsProject project, clsTask task)
        {
            project.TasksList.Add(task);
        }

        public static void AddProject(List<clsProject> projects, clsProject project)
        {
            projects.Add(project);
        }
        public static void AddWorkTime(clsTask Task, DateTime EndTime, DateTime StartTime)
        {
            if (Task == null) return;
            TimeSpan duration = EndTime - StartTime;
            if (duration.TotalSeconds > 0)
                Task.TotalTime += (int)duration.TotalSeconds;
        }

    }
}

using System;
using System.Collections.Generic;
using System.Drawing;
using System.IO;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace Personal_Project_Time_Tracker
{
    public static class AppData
    {

        private static bool IsInValidCharector(TextBox t)
        {

            return t.Text.Contains("#") || t.Text.Contains(":");
        }

        public static bool Validation(ErrorProvider ep, TextBox tName, TextBox tDescription, NumericUpDown nud = null, NumericUpDown nud2 = null)
        {
            // Clear Error Provider For Last Errors
            ep.Clear();

            // Validation  For Froject Name
            if (string.IsNullOrWhiteSpace(tName.Text))
            {
                ep.SetError(tName, "Required");
                return false;
            }
            if (IsInValidCharector(tName))
            {
                ep.SetError(tName, "The Fiad Name Cannon Contains : OR #");
                return false;
            }

            if (string.IsNullOrWhiteSpace(tDescription.Text))
            {
                ep.SetError(tDescription, "Required");
                return false;
            }
            if (IsInValidCharector(tDescription))
            {
                ep.SetError(tDescription, "The Faild Description Cannon Contains : OR #");
                return false;
            }

            if (nud != null)
                if (nud.Value < 0 & nud2.Value == 0)
                {
                    ep.SetError(nud, "The Zero Hours IS Invalid So We Change 0 to 1");
                    ep.Icon = SystemIcons.Warning;
                    nud.Value = 1;
                }
            if (nud2 != null && nud.Value == 0)
                if (nud2.Value < 0)
                {
                    ep.SetError(nud2, "The Zero Minutes IS Invalid So We Change 0 to 1");
                    ep.Icon = SystemIcons.Warning;
                    nud2.Value = 1;
                }
            return true;
        }


        public static List<clsProject> ProjectsList { get; set; } = new List<clsProject>();


    }
}
